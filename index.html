Quick Start

From install to recognition in 30 seconds.

pip install funasr
# Or latest: pip install git+https://github.com/modelscope/FunASR.git
from funasr import AutoModel

# Chinese ASR with speaker diarization
model = AutoModel(
    model="paraformer-zh",
    vad_model="fsmn-vad",
    punc_model="ct-punc",
    spk_model="cam++",
)
res = model.generate(input="meeting.wav", batch_size_s=300)

for sent in res[0]["sentence_info"]:
    print(f"[Speaker {sent['spk']}] {sent['text']}")
