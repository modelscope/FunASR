Features

Speech Recognition

🔧

Multi-GPU training with DeepSpeed, dynamic batching, and simple JSONL data format.

Quick Start

# Install
pip install funasr
# Or install from source for latest features
pip install git+https://github.com/modelscope/FunASR.git

# Basic ASR
from funasr import AutoModel

model = AutoModel(model="paraformer-zh", vad_model="fsmn-vad", punc_model="ct-punc")
res = model.generate(input="audio.wav")
print(res[0]["text"])

# With speaker diarization
model = AutoModel(model="paraformer-zh", vad_model="fsmn-vad",
                  punc_model="ct-punc", spk_model="cam++")
res = model.generate(input="meeting.wav")
for sent in res[0]["sentence_info"]:
    print(f"Speaker {sent['spk']}: {sent['text']}")
